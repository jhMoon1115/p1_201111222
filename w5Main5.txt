@startuml

title Activity Diagram 


start

:user input marks;


if(marks<=100 and marks>=90) then(yes)
    :grade is A;
elseif(80<=marks<90) then(yes)
    :grade is B;
elseif(60<=marks<80) then(yes)
    :grade is C;
elseif(40<=marks<60) then(yes)
    :grade is D;
elseif(0<=marks<40) then(yes)
    :grade is F;
else
    :print error;
endif

:print grade;

stop

@enduml
